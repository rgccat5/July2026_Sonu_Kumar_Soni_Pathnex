#Ansible - Install Docker
 - name: Install Docker
   hosts: all
   become: yes
   task: 
      - name: Install Docker
        yum:
          name: docker
          state: present
       -name: Start Docker
        service:
           name: docker
           state: started
           enabled: yes

#Shell Scrip -Check for Running Services
#! /bin/bash
services=("nginx" "docker" "httpd")
for service in "${services[@]}"
do
   if systemcctl is-actice  --quit $service; then
       echo "$service is running"
    else
       echo " $service is not runnning"
    fi
done
