     host: all
     become: yes
     task:
       -name: nginx
        state: present
      - name: Cofigure nginx as reverse proxy
        templete:
         src: reverse-proxy.conf.j2
         dest: /ets/nginx/conf.d/reverse-proxy.conf
      - name: Restart ngins service
      service:
        name: nginx
        state: restarted
        enabled: yes

 * GitLab CI/CD - Multi-Stage Deplyment with Helm

   stages:
    -build
    -push
    -deploy

    stage: build
    script:
      -docker build -t pathnex-web-app
 
   push:
     stage: push
     script:
      - docker login -u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD"
      - docker push pathnex-web-app
    
   deploy:
     stage: deploy
     script:
