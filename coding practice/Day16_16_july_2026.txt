  * Jenkinsfile - Deploy to Multi-Cluster Environments
  pipeline {
      agent any
      stages {
         stage('Build') {
         steps {
             echo 'Building Docker image...'
             docker.build('pathnex-web-app')
            }
         }
         stage('Test') {
         steps {
           echo 'Running tests....'
            }
         }
         stag('Deploy to kubernetes') {
           steps {
              script {
                if (env.CLUSTER == 'us-east') {
                 sh 'kubectl --context=us-east apply -f kubernets/deployment.yaml'
               } else {
                 sh 'kubectl --context=us-west apply -f kubernetes/deployment.yaml'
             }
        }
       }
     }
    }
